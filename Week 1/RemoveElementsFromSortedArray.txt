class Solution {
    public int removeDuplicates(int[] nums) {
        HashSet<Integer> hash=new HashSet<>();
        for(int i=0;i<nums.length;i++)
        {
            hash.add(nums[i]);
            nums[i]=0;
        }
        int j=0;
        List<Integer> list = new ArrayList<Integer>(hash);
        Collections.sort(list);
        for(int i=0;i<list.size();i++)
        {
            nums[j]=list.get(i);
            j++;
        }
        return j;
    }
}
